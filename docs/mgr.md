document.addEventListener("DOMContentLoaded", function() {
    
    // STEP 1: Define your terms and URLs
    const glossaryMappings = {
        "Ing. Ctislav Fiala, Ph.D.": "https://kps.fsv.cvut.cz/index.php?lmut=cz&part=people&id=10",
        "prof. Ing. Jiří Cajthaml, Ph.D.": "https://geomatics.fsv.cvut.cz/employees/jiri-cajthaml/",
        "prof. Dr. Ing. Karel Pavelka": "https://geomatics.fsv.cvut.cz/employees/karel-pavelka/",
    };
    
// STEP 2: Target ONLY elements with the "course" class
    const courseElements = document.querySelectorAll('.course');
    
    // STEP 3: Loop through each ".course" container found on the page
    courseElements.forEach(function(element) {

        // Grab all HTML inside the container, including any nested tags like <p> or <strong> 
        let content = element.innerHTML;
        
        // STEP 4: Loop through each term in your glossary dictionary
        for (const [term, url] of Object.entries(glossaryMappings)) {
            
            // We escape ALL special regex characters just to be 100% safe with punctuation
            const safeTerm = term.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
            
            // We replaced \b with (^|\W) and (?=\W|$).
            // (^|\W) ensures the phrase is preceded by the start of the string or a non-word character.
            // (?=\W|$) ensures it is followed by a non-word character or the end of the string.
            const regex = new RegExp(`(^|\\W)(${safeTerm})(?=\\W|$)`, 'gi');
            
            // Replace the text with an HTML link, preserving the original capitalization.
            // $1 puts back whatever character came right before your phrase (like a space or a HTML bracket).
            // $2 represents your exact glossary phrase to preserve its original capitalization.
            content = content.replace(regex, `$1<a href="${url}" target="_blank" rel="noopener noreferrer">$2</a>`);
        }
        
        // STEP 5: Update the container with the newly linked content
        element.innerHTML = content;
    });
});
